48  Token scan_body(nonstd::string_view close, Token::Kind closeKind, nonstd::string_view close_trim = nonstd::string_view(), bool trim = false) {
49  again:
50  // skip whitespace (except for \n as it might be a close)
51  if (tok_start >= m_in.size()) {
52  return make_token(Token::Kind::Eof);
53  }
54  char ch = m_in[tok_start];
55  if (ch == ' ' || ch == '\t' || ch == '\r') {
56  tok_start += 1;
57  goto again;
58  }
59 
60  // check for close
61  if (!close_trim.empty() && inja::string_view::starts_with(m_in.substr(tok_start), close_trim)) {
62  state = State::Text;
63  pos = tok_start + close_trim.size();
64  Token tok = make_token(closeKind);
65  skip_whitespaces_and_newlines();
66  return tok;
67  }
68 
69  if (inja::string_view::starts_with(m_in.substr(tok_start), close)) {
70  state = State::Text;
71  pos = tok_start + close.size();
72  Token tok = make_token(closeKind);
73  if (trim) {
74  skip_whitespaces_and_first_newline();
75  }
76  return tok;
77  }
78 
79  // skip \n
80  if (ch == '\n') {
81  tok_start += 1;
82  goto again;
83  }
84 
85  pos = tok_start + 1;
86  if (std::isalpha(ch)) {
87  minus_state = MinusState::Operator;
88  return scan_id();
89  }
90 
91  MinusState current_minus_state = minus_state;
92  if (minus_state == MinusState::Operator) {
93  minus_state = MinusState::Number;
94  }
95 
96  switch (ch) {
97  case '+':
98  return make_token(Token::Kind::Plus);
99  case '-':
100  if (current_minus_state == MinusState::Operator) {
101  return make_token(Token::Kind::Minus);
102  }
103  return scan_number();
104  case '*':
105  return make_token(Token::Kind::Times);
106  case '/':
107  return make_token(Token::Kind::Slash);
108  case '^':
109  return make_token(Token::Kind::Power);
110  case '%':
111  return make_token(Token::Kind::Percent);
112  case '.':
113  return make_token(Token::Kind::Dot);
114  case ',':
115  return make_token(Token::Kind::Comma);
116  case ':':
117  return make_token(Token::Kind::Colon);
118  case '(':
119  return make_token(Token::Kind::LeftParen);
120  case ')':
121  minus_state = MinusState::Operator;
122  return make_token(Token::Kind::RightParen);
123  case '[':
124  return make_token(Token::Kind::LeftBracket);
125  case ']':
126  minus_state = MinusState::Operator;
127  return make_token(Token::Kind::RightBracket);
128  case '{':
129  return make_token(Token::Kind::LeftBrace);
130  case '}':
131  minus_state = MinusState::Operator;
132  return make_token(Token::Kind::RightBrace);
133  case '>':
134  if (pos < m_in.size() && m_in[pos] == '=') {
135  pos += 1;
136  return make_token(Token::Kind::GreaterEqual);
137  }
138  return make_token(Token::Kind::GreaterThan);
139  case '<':
140  if (pos < m_in.size() && m_in[pos] == '=') {
141  pos += 1;
142  return make_token(Token::Kind::LessEqual);
143  }
144  return make_token(Token::Kind::LessThan);
145  case '=':
146  if (pos < m_in.size() && m_in[pos] == '=') {
147  pos += 1;
148  return make_token(Token::Kind::Equal);
149  }
150  return make_token(Token::Kind::Unknown);
151  case '!':
152  if (pos < m_in.size() && m_in[pos] == '=') {
153  pos += 1;
154  return make_token(Token::Kind::NotEqual);
155  }
156  return make_token(Token::Kind::Unknown);
157  case '\"':
158  return scan_string();
159  case '0':
160  case '1':
161  case '2':
162  case '3':
163  case '4':
164  case '5':
165  case '6':
166  case '7':
167  case '8':
168  case '9':
169  minus_state = MinusState::Operator;
170  return scan_number();
171  case '_':
172  case '@':
173  case '$':
174  minus_state = MinusState::Operator;
175  return scan_id();
176  default:
177  return make_token(Token::Kind::Unknown);
178  }
179  }

181  Token scan_id() {
182  for (;;) {
183  if (pos >= m_in.size()) {
184  break;
185  }
186  char ch = m_in[pos];
187  if (!std::isalnum(ch) && ch != '.' && ch != '/' && ch != '_' && ch != '-') {
188  break;
189  }
190  pos += 1;
191  }
192  return make_token(Token::Kind::Id);
193  }

195  Token scan_number() {
196  for (;;) {
197  if (pos >= m_in.size()) {
198  break;
199  }
200  char ch = m_in[pos];
201  // be very permissive in lexer (we'll catch errors when conversion happens)
202  if (!std::isdigit(ch) && ch != '.' && ch != 'e' && ch != 'E' && ch != '+' && ch != '-') {
203  break;
204  }
205  pos += 1;
206  }
207  return make_token(Token::Kind::Number);
208  }

210  Token scan_string() {
211  bool escape {false};
212  for (;;) {
213  if (pos >= m_in.size()) {
214  break;
215  }
216  char ch = m_in[pos++];
217  if (ch == '\\') {
218  escape = true;
219  } else if (!escape && ch == m_in[tok_start]) {
220  break;
221  } else {
222  escape = false;
223  }
224  }
225  return make_token(Token::Kind::String);

238  void skip_whitespaces_and_first_newline() {
239  if (pos < m_in.size()) {
240  while (pos < m_in.size() && (m_in[pos] == ' ' || m_in[pos] == '\t')) {
241  pos += 1;
242  }
243  }
244 
245  if (pos < m_in.size()) {
246  char ch = m_in[pos];
247  if (ch == '\n') {
248  pos += 1;
249  } else if (ch == '\r') {
250  pos += 1;
251  if (pos < m_in.size() && m_in[pos] == '\n') {

258  static nonstd::string_view clear_final_line_if_whitespace(nonstd::string_view text) {
259  nonstd::string_view result = text;
260  while (!result.empty()) {
261  char ch = result.back();
262  if (ch == ' ' || ch == '\t') {
263  result.remove_suffix(1);
264  } else if (ch == '\n' || ch == '\r') {
265  break;
266  } else {
267  return text;
268  }
269  }
270  return result;

280  void start(nonstd::string_view input) {
281  m_in = input;
282  tok_start = 0;
283  pos = 0;
284  state = State::Text;
285  minus_state = MinusState::Number;
286

293  Token scan() {
294  tok_start = pos;
295 
296  again:
297  if (tok_start >= m_in.size()) {
298  return make_token(Token::Kind::Eof);
299  }
300 
301  switch (state) {
302  default:
303  case State::Text: {
304  // fast-scan to first open character
305  size_t open_start = m_in.substr(pos).find_first_of(config.open_chars);
306  if (open_start == nonstd::string_view::npos) {
307  // didn't find open, return remaining text as text token
308  pos = m_in.size();
309  return make_token(Token::Kind::Text);
310  }
311  pos += open_start;
312 
313  // try to match one of the opening sequences, and get the close
314  nonstd::string_view open_str = m_in.substr(pos);
315  bool must_lstrip = false;
316  if (inja::string_view::starts_with(open_str, config.expression_open)) {
317  if (inja::string_view::starts_with(open_str, config.expression_open_force_lstrip)) {
318  state = State::ExpressionStartForceLstrip;
319  must_lstrip = true;
320  } else {
321  state = State::ExpressionStart;
322  }
323  } else if (inja::string_view::starts_with(open_str, config.statement_open)) {
324  if (inja::string_view::starts_with(open_str, config.statement_open_no_lstrip)) {
325  state = State::StatementStartNoLstrip;
326  } else if (inja::string_view::starts_with(open_str, config.statement_open_force_lstrip )) {
327  state = State::StatementStartForceLstrip;
328  must_lstrip = true;
329  } else {
330  state = State::StatementStart;
331  must_lstrip = config.lstrip_blocks;
332  }
333  } else if (inja::string_view::starts_with(open_str, config.comment_open)) {
334  state = State::CommentStart;
335  must_lstrip = config.lstrip_blocks;
336  } else if ((pos == 0 || m_in[pos - 1] == '\n') && inja::string_view::starts_with(open_str, config.line_statement)) {
337  state = State::LineStart;
338  } else {
339  pos += 1; // wasn't actually an opening sequence
340  goto again;
341  }
342 
343  nonstd::string_view text = string_view::slice(m_in, tok_start, pos);
344  if (must_lstrip) {
345  text = clear_final_line_if_whitespace(text);
346  }
347 
348  if (text.empty()) {
349  goto again; // don't generate empty token
350  }
351  return Token(Token::Kind::Text, text);
352  }
353  case State::ExpressionStart: {
354  state = State::ExpressionBody;
355  pos += config.expression_open.size();
356  return make_token(Token::Kind::ExpressionOpen);
357  }
358  case State::ExpressionStartForceLstrip: {
359  state = State::ExpressionBody;
360  pos += config.expression_open_force_lstrip.size();
361  return make_token(Token::Kind::ExpressionOpen);
362  }
363  case State::LineStart: {
364  state = State::LineBody;
365  pos += config.line_statement.size();
366  return make_token(Token::Kind::LineStatementOpen);
367  }
368  case State::StatementStart: {
369  state = State::StatementBody;
370  pos += config.statement_open.size();
371  return make_token(Token::Kind::StatementOpen);
372  }
373  case State::StatementStartNoLstrip: {
374  state = State::StatementBody;
375  pos += config.statement_open_no_lstrip.size();
376  return make_token(Token::Kind::StatementOpen);
377  }
378  case State::StatementStartForceLstrip: {
379  state = State::StatementBody;
380  pos += config.statement_open_force_lstrip.size();
381  return make_token(Token::Kind::StatementOpen);
382  }
383  case State::CommentStart: {
384  state = State::CommentBody;
385  pos += config.comment_open.size();
386  return make_token(Token::Kind::CommentOpen);
387  }
388  case State::ExpressionBody:
389  return scan_body(config.expression_close, Token::Kind::ExpressionClose, config.expression_close_force_rstrip);
390  case State::LineBody:
391  return scan_body("\n", Token::Kind::LineStatementClose);
392  case State::StatementBody:
393  return scan_body(config.statement_close, Token::Kind::StatementClose, config.statement_close_force_rstrip, config.trim_blocks);
394  case State::CommentBody: {
395  // fast-scan to comment close
396  size_t end = m_in.substr(pos).find(config.comment_close);
397  if (end == nonstd::string_view::npos) {
398  pos = m_in.size();
399  return make_token(Token::Kind::Eof);
400  }
401  // return the entire comment in the close token
402  state = State::Text;
403  pos += end + config.comment_close.size();
404  Token tok = make_token(Token::Kind::CommentClose);
405  if (config.trim_blocks) {
406  skip_whitespaces_and_first_newline();
407  }
408  return tok;
409  }
410  }
